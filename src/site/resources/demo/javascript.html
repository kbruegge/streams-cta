Demo - using JavaScript

JavaScript can be handy for quickly adding stuff to your process: 

   <container>
      <stream id="example" class="stream.io.CsvStream"
                 url="http://www.jwall.org/streams/demo.csv" />

      <process input="example">
         <JavaScript>
            function process(data){
              var x1 = data.get( "x1" ) * 1;
              var x2 = data.get( "x2" ) * 1;
              var x3 = data.get( "x3" ) * 1;

              var sum = x1 + x2 + x3;
              data.put( "x4", sum );
   
              return data;
            }
         </JavaScript>
         <PrintData />
      </process>
   </container>
