Demo - using the stream-plotter

Adding the stream-plotter package to plot the data. 

   <container>
      <libs>
          org.jwall:stream-plotter:0.9.5-SNAPSHOT
      </libs>

      <stream id="example" class="stream.io.CsvStream"
                 url="http://www.jwall.org/streams/demo.csv" />

      <process input="example">

         <stream.flow.Delay time="100ms"/>

         <stream.plotter.Plotter keys="x1,x2,x3" />

         <PrintData />
      </process>
   </container>
